  "rules": {
    // Rule 1: Make "movies" data publicly readable, but not writable.
    "movies": {
      ".read": true,
      ".write": "auth != null" // Only logged-in users can write new movie data
    },
    
    // Rule 2: Secure user-specific data in a "watchlists" node.
    "watchlists": {
      // The $uid is a wildcard that matches the currently authenticated user's ID.
      "$uid": {
        ".read": "$uid === auth.uid",
        ".write": "$uid === auth.uid"
      }
    },
    
    // Rule 3: User profiles. Allow public read but owner-only write.
    "users": {
      "$uid": {
        ".read": true,
        ".write": "$uid === auth.uid"
      }
    },

    // Default Deny: By not defining .read or .write rules at the root,
    // all other paths are inaccessible by default.
    ".read": false,
    ".write": false
  }
}


FIREBASE Firestore Rules 

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Rule 1: Publicly readable "movies" collection
    // Anyone can view the list of movies, but only authenticated users can add them.
    // No one can update or delete them through the client.
    match /movies/{movieId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if false; // Protects data integrity from client-side changes
    }

    // Rule 2: User-specific "watchlist"
    // Users can only access and modify their own watchlist.
    // The document ID in the "watchlists" collection must be the user's ID (uid).
    match /watchlists/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Rule 3: User profiles
    // A user can read anyone's profile (e.g., to see a username or profile picture).
    // However, a user can only create and update their own profile.
    match /users/{userId} {
      allow read: if true;
      allow create, update: if request.auth != null && request.auth.uid == userId;
      allow delete: if false; // Typically, you don't want users to delete their own profiles
    }

    // Default Deny: For any other collection not specified above, deny all access.
    // This is an important security practice.
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
